Offcanvas

slide in/out · 4 directions · ESC · backdrop · focus trap · focus return · external trigger · accessible

Right (default)

Right Panel

This panel slides in from the right. Great for cart, filters, or content detail.

Left

Left Panel

This panel slides in from the left. Great for sidebar navigation or a main menu.

Top

Top Panel

This panel drops down from the top. Great for a search bar or announcement banner.

Bottom

Bottom Panel

This panel slides up from the bottom. Great for mobile action sheets or confirmation dialogs.

External Trigger

How it works

  • Wrap in [data-wpui="offcanvas"]. Set the slide direction with data-side="right|left|top|bottom" — default is right.
  • Add data-open to the trigger button, data-backdrop to the backdrop overlay, and data-panel to the sliding panel element.
  • To open from anywhere on the page, give the wrapper an id and add data-offcanvas="your-id" to any external button.
  • Direction is handled entirely by CSS attribute selectors — the JS doesn't care which side. ESC closes, focus is trapped inside, and returns to the opener on close.

Get the code

<div data-wpui="offcanvas" data-side="right" class="wpui-offcanvas">
    <!-- data-side: "right" | "left" | "top" | "bottom" -->

    <button data-open class="wpui-btn">Open</button>

    <div class="wpui-offcanvas__backdrop" data-backdrop></div>

    <div class="wpui-offcanvas__panel" data-panel
        role="dialog"
        aria-modal="true"
        aria-hidden="true"
        aria-labelledby="panel-title">

        <div class="wpui-offcanvas__header">
            <h2 class="wpui-offcanvas__title" id="panel-title">Panel Title</h2>
            <button data-close class="wpui-offcanvas__close" aria-label="Close panel">&times;</button>
        </div>

        <div class="wpui-offcanvas__body">
            <p>Panel content goes here.</p>
        </div>

    </div>
</div>

(function () {

  const FOCUSABLE = [
    'a[href]',
    'button:not([disabled])',
    'textarea', 'input', 'select',
    '[tabindex]:not([tabindex="-1"])'
  ].join(',')

  const panels = {}

  const initOffcanvas = (root) => {

    const id = root.id
    const openBtn = root.querySelector('[data-open]')
    const closeBtn = root.querySelector('[data-close]')
    const backdrop = root.querySelector('[data-backdrop]')
    const panel = root.querySelector('[data-panel]')

    let focusable = []
    let first, last, opener

    const updateFocusable = () => {
      focusable = Array.from(panel.querySelectorAll(FOCUSABLE))
      first = focusable[0]
      last = focusable[focusable.length - 1]
    }

    const open = (trigger) => {
      opener = trigger || openBtn || document.activeElement
      root.classList.add('wpui-offcanvas--open')
      panel.setAttribute('aria-hidden', 'false')
      document.body.style.overflow = 'hidden'
      updateFocusable()
      setTimeout(() => first?.focus(), 50)
    }

    const close = () => {
      root.classList.remove('wpui-offcanvas--open')
      panel.setAttribute('aria-hidden', 'true')
      document.body.style.overflow = ''
      opener?.focus()
      opener = null
    }

    openBtn?.addEventListener('click', () => open(openBtn))
    closeBtn?.addEventListener('click', close)
    backdrop?.addEventListener('click', close)

    document.addEventListener('keydown', (e) => {
      if (!root.classList.contains('wpui-offcanvas--open')) return

      if (e.key === 'Escape') { close(); return }

      if (e.key === 'Tab') {
        updateFocusable()
        if (focusable.length === 0) { e.preventDefault(); return }

        if (e.shiftKey) {
          if (document.activeElement === first) { e.preventDefault(); last.focus() }
        } else {
          if (document.activeElement === last) { e.preventDefault(); first.focus() }
        }
      }
    })

    if (id) panels[id] = { open, close }
  }

  document.querySelectorAll('[data-wpui="offcanvas"]').forEach(initOffcanvas)

  // External trigger: <button data-offcanvas="panel-id">Open</button>
  document.addEventListener('click', (e) => {
    const trigger = e.target.closest('[data-offcanvas]')
    if (!trigger) return
    e.preventDefault()
    panels[trigger.dataset.offcanvas]?.open(trigger)
  })

})()
